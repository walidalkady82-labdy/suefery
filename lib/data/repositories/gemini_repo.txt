class GeminiRepo implements IGeminiRepo {
  final _log = LoggerRepo('GeminiRepo');
  final GenerativeModel _orderModel;
  final GenerativeModel _chefModel;
  final GenerativeModel _chatModel;

  GeminiRepo._({
    required GenerativeModel orderModel,
    required GenerativeModel chefModel,
    required GenerativeModel chatModel,
  })  : _orderModel = orderModel,
        _chefModel = chefModel,
        _chatModel = chatModel;

  factory GeminiRepo.create() {
    final generativeModel = FirebaseAI.googleAI().generativeModel(
        model: 'gemini-2.5-flash', 
        );

    // Configuration for the Order Assistant model
    final orderModel = FirebaseAI.googleAI().generativeModel(
      model: 'gemini-2.5-flash',
      systemInstruction: Content.system(_deliveryAppSystemPrompt),
      generationConfig: GenerationConfig.fromJson(_deliveryAppGenerationConfig),
    );

    // Configuration for the Chef model
    final chefModel = FirebaseAI.googleAI().generativeModel(
      model: 'gemini-2.5-flash',
      systemInstruction: Content.system(_chefSystemPrompt),
      generationConfig: GenerationConfig.fromJson(_chefGenerationConfig),
    );

    // Configuration for the generic Chat model
    final chatModel = generativeModel;

    return GeminiRepo._(
      orderModel: orderModel,
      chefModel: chefModel,
      chatModel: chatModel,
    );
  }

  @override
  Future<Map<String, dynamic>> generateOrderContent(
      List<Map<String, dynamic>> history) async {
    try {
      _log.i('Generating content with Order model...');
      final contents = history.map((item) {
        final role = item['role'] as String;
        final parts = item['parts'] as List<dynamic>;
        if (parts.isNotEmpty && parts[0]['text'] != null) {
          final text = parts[0]['text'] as String;
          return Content(role: role, parts: [TextPart(text)]);
        }
        return Content.text('');
      }).toList();

      final response = await _orderModel.generateContent(contents);

      final jsonText = response.text;
      if (jsonText == null) {
        _log.e('API Error: Response text is null or empty.');
        throw Exception('Failed to get response from AI: No text in response.');
      }
      return jsonDecode(jsonText) as Map<String, dynamic>;
    } catch (e) {
      _log.e('API Error in generateContent: $e');
      rethrow;
    }
  }

  @override
  Future<Map<String, dynamic>> generateRecipeContent(String prompt) async {
    try {
      _log.i('Generating content with Chef model...');
      final content = [Content.text(prompt)];
      final response = await _chefModel.generateContent(content);

      final jsonText = response.text;
      if (jsonText == null) {
        _log.e('API Error: Response text is null or empty.');
        throw Exception('Failed to get response from AI: No text in response.');
      }
      return jsonDecode(jsonText) as Map<String, dynamic>;
    } catch (e) {
      _log.e('API Error in generateRecipeContent: $e');
      rethrow;
    }
  }

  @override
  Future<String> generateText(
      {required String prompt, List<Map<String, dynamic>>? history}) async {
    try {
      _log.i('Generating text with Chat model...');

      // Convert history and current prompt into List<Content>
      final List<Content> contents = [];
      if (history != null) {
        for (var item in history) {
          final role = item['role'] as String;
          final parts = item['parts'] as List<dynamic>;
          if (parts.isNotEmpty && parts[0]['text'] != null) {
            final text = parts[0]['text'] as String;
            contents.add(Content(role, [TextPart(text)]));
          }
        }
      }
      contents.add(Content.user([TextPart(prompt)])); // Add the current user prompt

      final response = await _chatModel.generateContent(contents);

      if (response.text != null) {
        return response.text!;
      } else {
        throw Exception('Failed to get response from AI: No text in response.');
      }
    }
  }
}

const String _deliveryAppSystemPrompt = """
You are a helpful and friendly delivery assistant for a local service called "Suefery".
Your primary goal is to understand the user's order from their chat messages and confirm it with them.
When you are confident about the items and quantities, you MUST format your response as a JSON object.

The JSON object has two main keys:
1. "ai_response_text": A friendly, conversational string where you confirm the order with the user.
2. "parsed_order": An object containing the structured order details.

The "parsed_order" object has two keys:
1. "order_confirmed": A boolean. Set to `true` only when you have identified specific items to order. If the user is just chatting, set it to `false`.
2. "requested_items": A list of objects, where each object represents an item in the order.

Each item object has: "item_name" (string), "quantity" (integer), and can optionally have "notes" (string) and "unit_price" (double).

Example of a confirmed order:
{
  "ai_response_text": "You got it! That's 2 large bottles of water and 1 bag of spicy chips. Does that look right?",
  "parsed_order": {
    "order_confirmed": true,
    "requested_items": [
      {"item_name": "Water Bottle", "quantity": 2, "notes": "Large", "unit_price": 5.0},
      {"item_name": "Chips", "quantity": 1, "notes": "Spicy", "unit_price": 8.0}
    ]
  }
}

If the user is just asking a question or making small talk, do not try to create an order.
Example of a chat response:
{
  "ai_response_text": "We deliver anywhere in Beni Suef! What can I get for you today?",
  "parsed_order": {
    "order_confirmed": false,
    "requested_items": []
  }
}
""";
